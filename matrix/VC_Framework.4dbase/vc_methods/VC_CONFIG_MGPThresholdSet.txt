C_LONGINT($error_l)

$threshold_l:=$1

$error_l:=VC_EXT_PrefSet (<>VC_LABEL_DeleteMGPThreshold_t;String($threshold_l))

If ($error_l#0)
	UTIL_ERR_HandleError ($error_l;<>VC_ComponentName_t;Current method name)
Else 
	  // Make sure to update IP var.
	<>VCM_MGPThreshold_l:=$threshold_l
End if
